trade_action_tier_1_techs = {
	fire_and_forget = yes

	potential = {
		FROM = {
			NOT = {
				has_country_flag = tier_1_teched
			}
		}
		OR = {
			has_country_flag = old_nation
			is_country_type = stagehand
		}
		NOR = {
			has_modifier = old_country_1
			has_modifier = old_country_2
			has_modifier = old_country_3
		}
	}

	on_traded_effect = {
		FROM = {
			give_tier_1_techs_effect = yes
		}
	}

	ai_weight = {
		weight = 10
	}
}
